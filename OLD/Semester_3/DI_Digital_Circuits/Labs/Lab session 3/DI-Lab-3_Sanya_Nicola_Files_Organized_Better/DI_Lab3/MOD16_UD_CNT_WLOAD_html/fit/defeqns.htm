********** Mapped Logic **********
FDCPE_CNT0: FDCPE port map (CNT(0),CNT_D(0),CLK,NOT ARESETN,'0','1');
     CNT_D(0) <= ((LOAD AND PRE(0))
      OR (CNT(0) AND NOT LOAD AND CNTEN)
      OR (NOT CNT(0) AND NOT LOAD AND NOT CNTEN));
FTCPE_CNT1: FTCPE port map (CNT(1),CNT_T(1),CLK,NOT ARESETN,'0','1');
     CNT_T(1) <= ((LOAD AND CNT(1) AND NOT PRE(1))
      OR (LOAD AND NOT CNT(1) AND PRE(1))
      OR (CNT(0) AND NOT LOAD AND NOT CNTEN AND DIR)
      OR (NOT CNT(0) AND NOT LOAD AND NOT CNTEN AND NOT DIR));
FTCPE_CNT2: FTCPE port map (CNT(2),CNT_T(2),CLK,NOT ARESETN,'0','1');
     CNT_T(2) <= ((LOAD AND CNT(2) AND NOT PRE(2))
      OR (LOAD AND NOT CNT(2) AND PRE(2))
      OR (CNT(0) AND NOT LOAD AND NOT CNTEN AND DIR AND CNT(1))
      OR (NOT CNT(0) AND NOT LOAD AND NOT CNTEN AND NOT DIR AND NOT CNT(1)));
FTCPE_CNT3: FTCPE port map (CNT(3),CNT_T(3),CLK,NOT ARESETN,'0','1');
     CNT_T(3) <= ((LOAD AND CNT(3) AND NOT PRE(3))
      OR (LOAD AND NOT CNT(3) AND PRE(3))
      OR (CNT(0) AND NOT LOAD AND NOT CNTEN AND DIR AND CNT(1) AND CNT(2))
      OR (NOT CNT(0) AND NOT LOAD AND NOT CNTEN AND NOT DIR AND NOT CNT(1) AND NOT CNT(2)));
TC <= ((CNT(0) AND DIR AND CNT(1) AND CNT(2) AND CNT(3) AND
      NOT TCEN)
      OR (NOT CNT(0) AND NOT DIR AND NOT CNT(1) AND NOT CNT(2) AND NOT CNT(3) AND
      NOT TCEN));
